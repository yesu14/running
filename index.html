alter table fitness_runs disable row level security;
alter table fitness_jumps disable row level security;
또는 RLS 정책 추가:
create policy "anon_all" on fitness_runs for all to anon using (true) with check (true);
create policy "anon_all" on fitness_jumps for all to anon using (true) with check (true);
